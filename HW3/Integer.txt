package edu.softserve;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Numbers2 {

	public static void main(String[] args)  throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		System.out.println("What is g?");
		int g = Integer.parseInt(br.readLine());
		System.out.println("What is h?");
		int h = Integer.parseInt(br.readLine());
		System.out.println("What is i?");
		int i = Integer.parseInt(br.readLine());

		if (g > h & g > i)
			System.out.println("Max number is: " + g);
		else if (h > g & h > i)
			System.out.println("Max number is: " + h);
		else
			System.out.println("Max number is: " + i);
		
		if (g < h & g < i)
			System.out.println("Min number is: " + g);
		else if (h < g & h < i)
			System.out.println("Min number is: " + h);
		else
			System.out.println("Min number is: " + i);
		

	}

}
